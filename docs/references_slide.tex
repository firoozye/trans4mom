\section{References and Further Reading}

\begin{frame}[allowframebreaks]{Primary Academic References}
    \begin{itemize}
        \item \textbf{Wood, K., G Geigerich, S., Roberts, S., \& Zohren, S. (2022).} \textit{Trading with the Momentum Transformer: An Intelligent and Interpretable Architecture.} Journal of Financial Data Science. \\
        \textit{Context:} The foundational architecture for this masterclass, introducing the VSN-Transformer hybrid.
        
        \item \textbf{Lim, B., Zohren, S., \& Roberts, S. (2019).} \textit{Enhancing Time-Series Momentum Strategies Using Deep Neural Networks.} Journal of Financial Data Science. \\
        \textit{Context:} The first major application of Sharpe-optimized LSTMs to multi-asset trend following.
        
        \item \textbf{Zohren, S. (2024).} \textit{Deep Learning in Finance.} Oxford University Press. \\
        \textit{Context:} A comprehensive textbook covering the state-of-the-art in neural architectures for financial time series.
        
        \item \textbf{Abdi, F., \& Ranaldo, A. (2017).} \textit{A Simple Estimation of Bid-Ask Spreads from Daily Close, High, and Low Prices.} Review of Financial Studies. \\
        \textit{Context:} The source of our dynamic liquidity friction estimator.
        
        \item \textbf{Lemp\'eri\`ere, Y., et al. (2014).} \textit{Two Centuries of Trend Following.} Journal of Investment Strategies. \\
        \textit{Context:} Archaeological proof of the persistent momentum risk premium across two centuries.
    \end{itemize}
\end{frame}
