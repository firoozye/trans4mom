\section{Part II: Evolution of Neural Architectures for Sequential Data}

\begin{frame}{Deep Momentum Networks (DMN): The LSTM Foundation}
    Lim, Zohren, and Roberts (2019) introduced the first end-to-end Sharpe-optimized framework using Recurrent Neural Networks (LSTMs).
    \begin{itemize}
        \item \textbf{Recurrent Logic:} Maintains a hidden state $h_t$ as a compact summary of past information.
        \item \textbf{Memory Bottleneck:} Despite gating, LSTMs struggle to preserve precise information over very long horizons (e.g., 252 days).
        \item \textbf{Vanishing Gradients:} Backpropagation through time (BPTT) remains challenging for extremely long dependencies.
    \end{itemize}
\end{frame}

\begin{frame}{The Architectural Leap: From Recurrence to Attention}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{LSTM (Sequential)}
            \begin{itemize}
                \item Path Length: $O(T)$
                \item Memory: Compressed cell state
                \item Bottleneck: Sequential dependency
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{Transformer (Parallel)}
            \begin{itemize}
                \item Path Length: $O(1)$
                \item Memory: Direct access to all $T$
                \item Interpretability: Attention Maps
            \end{itemize}
        \end{column}
    \end{columns}
    \textit{Insight:} The Transformer effectively eliminates the "Gating Gap" by allowing today's position to attend directly to a specific macro regime 252 days in the past.
\end{frame}

\section{Part III: The Momentum Transformer: Structural Innovations}

\begin{frame}{Variable Selection Networks (VSN)}
    The VSN automates feature selection, assigning importance weights $v$ to each input:
    \begin{equation}
        \tilde{x}_t = \sum_{i=1}^m v_t^{(i)} \cdot \tilde{x}_t^{(i)}
    \end{equation}
    \begin{itemize}
        \item Allows the model to dynamically prioritize specific MACD scales or asset classes.
        \item Effectively filters out uninformative or noisy inputs in low signal-to-noise environments.
    \end{itemize}
\end{frame}

\begin{frame}{Residual Learning: Constraining High-Capacity Models}
    To prevent unconstrained overfitting, we adopt a \textbf{Residual Learning} approach. The Transformer modulates a robust empirical prior:
    \begin{equation}
        X_t = \tanh(\alpha_t \cdot \text{Prior}_t)
    \end{equation}
    \begin{itemize}
        \item \textbf{The "Handcuffs" Principle:} A Transformer is a high-capacity learner. By forcing it to scale a known prior, we prevent it from "inventing" trades from noise.
        \item \textbf{Guided Learning:} The model focuses on the \textit{residual} alpha---the intelligent sizing and timing of a proven trend-following rule.
    \end{itemize}
\end{frame}

\section{Part IV: Empirical Rigor and Robustness}

\begin{frame}{Robustness via Multi-Seed Ensembling}
    Financial deep learning exhibits significant variance across random initializations.
    \begin{itemize}
        \item \textbf{Initialization Sensitivity:} Small differences in starting weights can lead to vastly different OOS outcomes.
        \item \textbf{Ensemble Strategy:} We train multiple models (e.g., 5 seeds) and average their position signals.
        \item \textbf{Statistical Stability:} Ensembling reduces the standard deviation of results and mitigates the impact of "outlier" overfitted runs.
    \end{itemize}
\end{frame}
