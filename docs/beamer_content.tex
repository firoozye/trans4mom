\begin{frame}{Failure Analysis: The Saturation Trap}
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{The Symptom:}
            \begin{itemize}
                \item Training Loss pinned at exactly 0.1064.
                \item Predictions stuck at constant -1.000.
                \item Zero gradient flow.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{The Diagnosis:}
            \begin{itemize}
                \item \textbf{High Learning Rate (0.001)} exploded weights early.
                \item \textbf{Tanh Saturation:} Large weights pushed activation to the limit.
                \item \textbf{Numerical Instability} in raw Sharpe Loss.
            \end{itemize}
        \end{column}
    \end{columns}

% --- SLIDE 3: Architectural Stability ---
\begin{frame}{Engineering for Stability: The Fixes}
    How we stabilized the Momentum Transformer:
    \begin{enumerate}
        \item \textbf{LayerNorm Integration:} Normalised features before the Tanh head to prevent saturation.
        \item \textbf{Per-Batch Sharpe:} Calculated Sharpe per-sample before averaging to maintain signal variance.
        \item \textbf{Learning Rate Pivot:} Reduced LR by 10x ($10^{-3} \to 10^{-4}$) for precise convergence.
        \item \textbf{Causal Masking:} Removed look-ahead bias to ensure realistic performance.
    \end{enumerate}
\end{frame}

% --- SLIDE 7: Senior Engineer Lessons ---
\begin{frame}{Closing: Lessons from the Cluster}
    \begin{itemize}
        \item \textbf{Diversification is the Only Free Lunch:} Boosting from 5 to 15 assets more than doubled the Sharpe Ratio (1.14 $\to$ 2.74).
        \item \textbf{Pessimistic Training:} Training with 100bps friction creates a "Fortress Strategy" that thrives in realistic 20bps environments.
        \item \textbf{The Stability Pivot:} Daily data and market-neutral constraints fixed the overtrading disaster of the 4h crypto model.
        \item \textbf{Infrastructure Matters:} Seamlessly scaling from local "toy" runs to 15-asset cluster jobs is the hallmark of a production pipeline.
    \end{itemize}
\end{frame}
